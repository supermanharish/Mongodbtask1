10.Delete the products which product price value are same
db.products.aggregate([ 
    {$group:{ 
    _id:"$product_price",
    duplicate:{$addToSet:"$_id"},
    count:{$sum:1}
    }},
      {
        $match : {
        count : {$gt:1}
    }
}
]).forEach ((doc)=>{
    doc.duplicate.shift();
    db.products.deleteMany({_id:{$in:doc.duplicate}})
})
